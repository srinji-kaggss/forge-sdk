5/// The execution contract for an agent run.
6///
7/// Carries all parameters that define the boundaries of an agent execution,
8/// including task description, resource limits, environment variables, and
9/// identity tracking via trace/run/session IDs.
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct AgentContext {
12    /// The task description for the agent to execute.
13    pub task: String,
14    /// The current working directory for file operations.
15    pub cwd: String,
16    /// Maximum number of steps the agent may take.
17    pub max_steps: u32,
18    /// Maximum number of tokens the agent may consume.
19    pub max_tokens: u64,
20    /// Maximum cost in USD the agent may incur.
21    pub max_cost: f64,
22    /// Trace identifier for observability correlation.
23    pub trace_id: String,
24    /// Run identifier scoped to a single execution.
25    pub run_id: String,
26    /// Optional session identifier for multi-run sessions.
27    pub session_id: Option<String>,
28    /// The model identifier (e.g. "gemini-2.0-flash").
29    pub model: String,
30    /// The provider identifier (e.g. "gemini", "ollama").
31    pub provider: String,
32    /// Environment variables available to the agent (keys → values).
33    pub env_vars: HashMap<String, String>,
34}
35
36impl AgentContext {
37    /// Return a new `AgentContext` with all required fields.
38    #[allow(clippy::too_many_arguments)]
39    pub fn new(
40        task: impl Into<String>,
41        cwd: impl Into<String>,
42        max_steps: u32,
43        max_tokens: u64,
44        max_cost: f64,
45        trace_id: impl Into<String>,
46        run_id: impl Into<String>,
47        session_id: Option<String>,
48        model: impl Into<String>,
49        provider: impl Into<String>,
50        env_vars: HashMap<String, String>,
51    ) -> Self {
52        Self {
53            task: task.into(),
54            cwd: cwd.into(),
55            max_steps,
56            max_tokens,
57            max_cost,
58            trace_id: trace_id.into(),
59            run_id: run_id.into(),
60            session_id,
61            model: model.into(),
62            provider: provider.into(),
63            env_vars,
64        }
65    }
66
67    /// Truncate the task description to at most `max_chars` characters.
68    ///
69    /// Uses `.chars().take(max_chars)` to respect Unicode grapheme boundaries
70    /// rather than byte-slicing with `text[..n]`.
71    pub fn char_count_aware_truncation(&mut self, max_chars: usize) {
72        let truncated: String = self.task.chars().take(max_chars).collect();
73        self.task = truncated;
74    }
75}
